T removeAt(int index, {bool notifyChanges = true}) {
  final removedValue = _value.removeAt(index);

  if (notifyChanges) {
    _viewModel.notifyChanges([EmpireStateChanged.removedFromList(removedValue)]);
  }

  return removedValue;
}
